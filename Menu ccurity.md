document.addEventListener('DOMContentLoaded', function () {
  var brandLinks = document.querySelectorAll('.cc-brand-logo-link');

  brandLinks.forEach(function (link) {
    var domain = link.getAttribute('data-domain');
    if (!domain) return;

    var iconContainer = link.querySelector('.cc-brand-logo-favicon');
    if (!iconContainer) return;

    var img = document.createElement('img');
    // Usamos el servicio de Google para obtener el favicon del dominio
    img.src = 'https://www.google.com/s2/favicons?domain=' +
              encodeURIComponent(domain) + '&sz=32';
    img.alt = '';
    img.loading = 'lazy';

    iconContainer.appendChild(img);
  });
});

<script>
(function () {
  const menu = document.getElementById('menu');
  const menuButton = document.getElementById('menu-button');

  function isMenuOpen() {
    return menu.classList.contains('menu-open');
  }

  function openMenu() {
    menu.classList.add('menu-open');
    menuButton.setAttribute('aria-expanded', 'true');
    menu.setAttribute('aria-hidden', 'false');
  }

  function closeMenu() {
    menu.classList.remove('menu-open');
    menuButton.setAttribute('aria-expanded', 'false');
    menu.setAttribute('aria-hidden', 'true');
  }

  window.toggleMenu = function () {
    if (!menu) return;
    if (isMenuOpen()) {
      closeMenu();
    } else {
      openMenu();
    }
  };

  // Cerrar menú al hacer clic fuera
  document.addEventListener('click', function (event) {
    if (!menu || !menuButton) return;
    const clickInsideMenu = menu.contains(event.target);
    const clickOnButton = menuButton.contains(event.target);

    if (!clickInsideMenu && !clickOnButton && isMenuOpen()) {
      closeMenu();
    }
  });

  // Cerrar menú al presionar la tecla Escape
  document.addEventListener('keydown', function (event) {
    if (event.key === 'Escape' && isMenuOpen()) {
      closeMenu();
      menuButton.focus(); // Retornar el foco al botón por accesibilidad
    }
  });

  // Asegurar que el menú arranque cerrado
  window.addEventListener('load', function () {
    closeMenu();
  });
})();
</script>
